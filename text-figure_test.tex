\documentclass{article}
\usepackage{varwidth}
\usepackage{graphicx}
\usepackage{lipsum}


\begin{document}
\ExplSyntaxOn
\coffin_new:N \l_test_a_coffin
\coffin_new:N \l_test_b_coffin
\hcoffin_set:Nn \l_test_b_coffin
  {
    \begin{varwidth}{\hsize}
      \lipsum[1]
      \begin{itemize}
        \item what~I~want
      \end{itemize}
    \end{varwidth}
  }
\hcoffin_set:Nn \l_tmpa_coffin
  {
    \begin{varwidth}{\hsize}
      \includegraphics[width = 2cm]{example-image-a}
    \end{varwidth}
  }
\coffin_join:NnnNnnnn
  \l_test_b_coffin { b } { r }
  \l_tmpa_coffin { t } { r }
  { 0pt } { 0pt }
\begin{enumerate}
  \item 
  \coffin_typeset:Nnnnn 
    \l_test_b_coffin { t } { l }
    { 0pt } { 0.5\baselineskip }
  \item \mode_leave_vertical:
    % \begin{varwidth}{\hsize}
      \hcoffin_set:Nn \l_test_a_coffin
    {
      \begin{varwidth}{\hsize}
        \lipsum[1]
        \begin{itemize}
          \item what~I~don't~want!
        \end{itemize}
      \end{varwidth}
    }
  \hcoffin_set:Nn \l_tmpa_coffin
    {
      \begin{varwidth}{\hsize}
        \includegraphics[width = 2cm]{example-image-a}
      \end{varwidth}
    }
  \coffin_join:NnnNnnnn
    \l_test_a_coffin { b } { r }
    \l_tmpa_coffin { t } { r }
    { 0pt } { 0pt }
  \coffin_typeset:Nnnnn 
    \l_test_a_coffin { t } { l }
    { 0pt } { 0.5\baselineskip }
\end{enumerate}
\ExplSyntaxOff
\end{document}
